%%% Themes: Awesome-CV
\documentclass[11pt, a4paper]{awesome-cv}

%%% Override a directory location for fonts(default: 'fonts/')
\fontdir[fonts/]

%%% Configure a directory location for sections
\newcommand*{\sectiondir}{cv/}

%%% Override color
\colorlet{awesome}{darkgray}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%     3rd party packages
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%% Needed to divide into several files
\usepackage{import}
\usepackage{csquotes}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%     Personal Data
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%% Essentials

\address{<111 Your Street>\enskip\cdotp\enskip <City>, <State>-<ZIP>\enskip\cdotp\enskip <Country>}
\email{<your.email@uni.edu>}
\mobile{<+1 000 000-0000>} 
\homepage{<your-homepage.com>}
\github{<GitHub name>}
\linkedin{<LinkedIn name>}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%     Content
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%% Make a footer for CV with three arguments(<left>, <center>, <right>)
\makecvfooter
  {\today}
  {Curriculum Vitae\enskip\cdotp\enskip <Your Name>}
  {Page \thepage/\pageref{LastPage}}

\begin{document}

\makecvheader
